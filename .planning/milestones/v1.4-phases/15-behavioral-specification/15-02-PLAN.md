export interface BehaviorSpec {
  tools: ToolBehavior[];
  workflows: WorkflowScenario[];
  antiPatterns: AntiPattern[];
  qualityCriteria: QualityCriterion[];
}

export interface ToolBehavior {
  name: string;          // e.g., "twining_decide"
  tier: 1 | 2;
  context?: string;
  rules: BehaviorRule[];
  correctUsage?: CodeExample;
  incorrectUsage?: CodeExample;
  incorrectReason?: string;
}

export interface BehaviorRule {
  id: string;            // e.g., "DECIDE-01"
  level: 'MUST' | 'SHOULD' | 'MUST_NOT';
  rule: string;
}

export interface WorkflowScenario {
  name: string;
  steps: WorkflowStep[];
}

export interface WorkflowStep {
  order: number;
  tool: string;
  purpose: string;
}

export interface AntiPattern {
  id: string;
  description: string;
  badExample: string;
  goodExample: string;
}

export interface QualityCriterion {
  name: string;
  levels: QualityLevel[];
}

export interface QualityLevel {
  level: string;
  description: string;
  example: string;
}

// Zod schemas also exported: BehaviorSpecSchema, etc.

Export a single function: `parseBehaviors(markdown: string): BehaviorSpec`

Implementation approach -- line-by-line state machine:
1. Split markdown into lines
2. Track state: current section (tool-behaviors, workflows, anti-patterns, quality-criteria), current subsection (context, rules, correct-usage, incorrect-usage), current tool name, current tier
3. Detect sections by heading level:
   - `## Tool Behaviors` -> tool-behaviors mode
   - `## Workflows` -> workflows mode
   - `## Anti-Patterns` -> anti-patterns mode
   - `## Quality Criteria` -> quality-criteria mode
4. Within tool-behaviors:
   - `### twining_*` -> new tool entry
   - `<!-- tier: N -->` -> set tier for current tool
   - `#### Context` -> capture prose until next ####
   - `#### Rules` -> parse markdown table (ID | Level | Rule)
   - `#### Correct Usage` -> capture code block
   - `#### Incorrect Usage` -> capture code block
   - `**Why incorrect:**` -> capture explanation
5. Within workflows:
   - `### workflow: name` -> new workflow
   - Table rows -> WorkflowStep entries
6. Within anti-patterns:
   - `### anti-pattern: id` -> new anti-pattern
   - `**Description:**` -> description text
   - `**Bad:**` -> capture until `**Good:**`
   - `**Good:**` -> capture until next ### or ##
7. Within quality-criteria:
   - `### quality: name` -> new criterion
   - Table rows -> QualityLevel entries

Use helper functions:
- `parseMarkdownTable(lines, startIdx)` -> { headers, rows, endIdx } (from RESEARCH code example)
- `extractCodeBlock(lines, startIdx)` -> { code, language, endIdx }
- `collectProse(lines, startIdx, stopPattern)` -> { text, endIdx }

Be tolerant of blank lines between sections. Use strict error reporting with line numbers for unexpected format issues (log warnings to stderr, don't throw).

Do NOT use any markdown AST library (no unified, no remark). This is a format-specific extractor for a controlled document, not a general markdown parser.
